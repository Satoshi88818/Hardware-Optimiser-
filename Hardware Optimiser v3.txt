"""
hardware_optimiser_v3.py
========================
Abstract, domain-agnostic hardware design optimiser — v3.

NEW in v3 (vs v2):
  §6b  Analytical-gradient cache — SLSQP uses exact Jacobians for nozzle & motor
  §7b  Strict feasibility post-processor + repair step (project back to feasible region)
  §8b  Parallel evaluator wrapper (concurrent.futures ProcessPoolExecutor, 4-20× speedup)
  §10b Pluggable mass-model registry (JSON tables or empirical regressions)
  §18  Streamlit dashboard — non-coders adjust bounds/weights/tolerances, see live trade-off plots

Supported domains  (--domain):
  nozzle            – de Laval rocket / supersonic nozzle
  electric_motor    – surface-mount PMSM sizing model
  centrifugal_pump  – single-stage centrifugal pump impeller

Optimisation modes  (--mode):
  single            – SLSQP from default / config start point
  monte_carlo       – Latin-Hypercube multi-start + SLSQP
  sobol             – Sobol quasi-random multi-start + SLSQP
  differential_evo  – SciPy differential evolution (global) then SLSQP polish

Post-optimisation analysis:
  --tolerance-mc N  – tolerance / manufacturing Monte-Carlo around the best
                      design (perturbs parameters by gaussian noise, reports
                      pass rate and KPI statistics)

Config file  (JSON or YAML, --config FILE):
  params:      { key: value }     – fixed / initial parameter overrides
  bounds:      { key: [lo, hi] }  – design-variable bound overrides
  weights:     { kpi_name: w }    – objective weight overrides
  tols:        { param: sigma_frac } – tolerance sigma as fraction of nominal
  mass_models: { domain: { model_type: "table"|"regression", ... } }

Run examples
------------

  python hardware_optimiser_v3.py --domain nozzle --mode sobol --n-runs 64 --seed 7
  python hardware_optimiser_v3.py --domain centrifugal_pump --mode differential_evo
  python hardware_optimiser_v3.py --domain electric_motor --tolerance-mc 2000 --seed 0
  python hardware_optimiser_v3.py --domain nozzle --parallel --workers 8
  python hardware_optimiser_v3.py --dashboard          # launches Streamlit UI
  python hardware_optimiser_v3.py --test

@dataclass
class DomainConfig:
    params:      Dict[str, float]               = field(default_factory=dict)
    bounds:      Dict[str, Tuple[float, float]] = field(default_factory=dict)
    weights:     Dict[str, float]               = field(default_factory=dict)
    tols:        Dict[str, float]               = field(default_factory=dict)
    mass_models: Dict[str, Any]                 = field(default_factory=dict)

    @classmethod
    def empty(cls) -> "DomainConfig":
        return cls()

    @classmethod
    def from_file(cls, path: str) -> "DomainConfig":
        p = Path(path)
        if not p.exists():
            raise FileNotFoundError(f"Config not found: {path}")
        if p.suffix in (".yaml", ".yml"):
            if not _YAML_AVAILABLE:
                raise ImportError("PyYAML required. pip install pyyaml")
            with p.open() as fh:
                raw: Dict[str, Any] = yaml.safe_load(fh) or {}
        elif p.suffix == ".json":
            with p.open() as fh:
                raw = json.load(fh)
        else:
            raise ValueError(f"Unsupported config format: {p.suffix}")

        def _fdict(section: str) -> Dict[str, float]:
            out: Dict[str, float] = {}
            for k, v in (raw.get(section) or {}).items():
                try:
                    out[str(k)] = float(v)
                except (TypeError, ValueError) as exc:
                    raise ValueError(f"Config {section}['{k}'] must be numeric.") from exc
            return out

        bounds: Dict[str, Tuple[float, float]] = {}
        for k, v in (raw.get("bounds") or {}).items():
            try:
                lo, hi = float(v[0]), float(v[1])
            except Exception as exc:
                raise ValueError(f"Config bounds['{k}'] must be [lo,hi].") from exc
            if lo >= hi:
                raise ValueError(f"Config bounds['{k}']: lo must be < hi.")
            bounds[str(k)] = (lo, hi)

        return cls(
            params=_fdict("params"),
            bounds=bounds,
            weights=_fdict("weights"),
            tols=_fdict("tols"),
            mass_models=raw.get("mass_models") or {},
        )

    @classmethod
    def from_file_or_empty(cls, path: Optional[str]) -> "DomainConfig":
        return cls.from_file(path) if path is not None else cls.empty()

    return {f.name: getattr(obj, f.name) for f in fields(obj)}

PhysicsFunc = Callable[[Dict[str, float]], float]


class PhysicsRegistry:
    def __init__(self, domain_name: str) -> None:
        self._name = domain_name
        self._reg: Dict[str, PhysicsFunc] = {}

    def register(self, quantities: Dict[str, PhysicsFunc]) -> None:
        dup = set(quantities) & set(self._reg)
        if dup:
            log.warning("PhysicsRegistry[%s]: overwriting %s.", self._name, dup)
        self._reg.update(quantities)

    def evaluate(self, quantity: str, subs: Dict[str, float]) -> float:
        if quantity not in self._reg:
            raise KeyError(f"PhysicsRegistry[{self._name}]: '{quantity}' not found. "
                           f"Available: {sorted(self._reg)}")
        return self._reg[quantity](subs)

    def __repr__(self) -> str:
        return f"PhysicsRegistry({self._name!r}, {sorted(self._reg)})"

class AbstractAxioms(ABC):
    @abstractmethod
    def evaluate(self, quantity: str, subs: Dict[str, float]) -> float: ...


class AbstractConstraints(ABC):
    @abstractmethod
    def get_funcs(self) -> List[Callable[[np.ndarray, Dict[str, float]], float]]:
        """Return a list of constraint functions g(x,base)>=0 for SLSQP."""
        ...


class AbstractGeometryGenerator(ABC):
    @abstractmethod
    def generate_geometry(self, params: Dict[str, float]) -> Dict[str, float]: ...
    @abstractmethod
    def calculate_mass(self, geo: Dict[str, float], params: Dict[str, float]) -> float: ...
    @abstractmethod
    def export_geometry(self, geo: Dict[str, float]) -> str: ...


class AbstractSimulator(ABC):
    @abstractmethod
    def simulate(self, params: Dict[str, float]) -> Dict[str, float]: ...


class AbstractEvaluator(ABC):
    @abstractmethod
    def evaluate(self, params: Dict[str, float]) -> Dict[str, Any]: ...

    @property
    @abstractmethod
    def tolerance_params(self) -> List[str]: ...

    def gradient(self, quantity: str, params: Dict[str, float],
                 design_var_names: List[str]) -> Optional[np.ndarray]:
        """Return analytical gradient of `quantity` wrt design vars, or None to fall back to FD."""
        return None


class AbstractOptimizer(ABC):
    @abstractmethod
    def optimize_design(self, initial_params: Dict[str, float],
                        x0_override: Optional[np.ndarray] = None) -> "OptimResult": ...


@dataclass
class OptimResult:
    optimized_params: Dict[str, float]
    results:          Dict[str, Any]
    objective_value:  float
    success:          bool
    message:          str
    n_iter:           int
    elapsed_s:        float
    feasible:         bool = True        # NEW: strict feasibility flag

    def scalar_results(self) -> Dict[str, float]:
        return {k: float(v) for k, v in self.results.items()
                if isinstance(v, (int, float)) and k != "geometry"}


class HardwareDomain(ABC):
    def __init__(self, config: DomainConfig) -> None:
        self._config = config

    @property
    @abstractmethod
    def axioms(self) -> AbstractAxioms: ...
    @property
    @abstractmethod
    def constraints(self) -> AbstractConstraints: ...
    @property
    @abstractmethod
    def geometry_generator(self) -> AbstractGeometryGenerator: ...
    @property
    @abstractmethod
    def evaluator(self) -> AbstractEvaluator: ...
    @property
    @abstractmethod
    def optimizer(self) -> AbstractOptimizer: ...
    @property
    @abstractmethod
    def design_vars(self) -> List[Tuple[str, Tuple[float, float]]]: ...
    @property
    @abstractmethod
    def constraint_funcs(self) -> List[Callable[[np.ndarray, Dict[str, float]], float]]: ...
    @property
    @abstractmethod
    def default_weights(self) -> Dict[str, float]: ...
    @property
    @abstractmethod
    def default_params(self) -> Dict[str, float]: ...

    def effective_params(self) -> Dict[str, float]:
        # Validate config keys against known params
        known = set(self.default_params.keys())
        unknown = set(self._config.params.keys()) - known
        if unknown:
            log.warning("Config params contain unknown keys for this domain: %s", unknown)
        return {**self.default_params, **self._config.params}

    def effective_design_vars(self) -> List[Tuple[str, Tuple[float, float]]]:
        return [(n, self._config.bounds.get(n, b)) for n, b in self.design_vars]

    def effective_weights(self) -> Dict[str, float]:
        return {**self.default_weights, **self._config.weights}

    def default_tols(self) -> Dict[str, float]:
        return {}

    @abstractmethod
    def _build_objective(self) -> "WeightedObjective": ...

# §6  WEIGHTED / NORMALISED OBJECTIVE FRAMEWORK

@dataclass
class ObjectiveTerm:
    kpi_key:    str
    weight:     float
    nominal:    float                        = 1.0
    constraint: Optional[Tuple[str, float]] = None   # ("ge"|"le", bound)
    penalty_k:  float                        = 100.0


class WeightedObjective:
    """
    Normalised weighted objective — no hand-tuned scale factors.

    Each term contributes:
        w_i * (result[kpi_i] - nominal_i) / |nominal_i|

    Soft constraint penalty added when violated:
        penalty_k * |violation_fraction|

    Weights may be overridden per-run via DomainConfig.weights.
    """

    def __init__(self, terms: List[ObjectiveTerm],
                 weight_overrides: Dict[str, float]) -> None:
        self._terms: List[ObjectiveTerm] = []
        for t in terms:
            w = weight_overrides.get(t.kpi_key, t.weight)
            self._terms.append(ObjectiveTerm(t.kpi_key, w, t.nominal, t.constraint, t.penalty_k))

    def __call__(self, results: Dict[str, Any]) -> float:
        total = 0.0
        for t in self._terms:
            val = results.get(t.kpi_key, 0.0)
            denom = abs(t.nominal) if t.nominal != 0.0 else 1.0
            total += t.weight * (val - t.nominal) / denom
            if t.constraint is not None:
                sense, bound = t.constraint
                denom_c = abs(bound) if bound != 0.0 else 1.0
                if sense == "ge" and val < bound:
                    total += t.penalty_k * (bound - val) / denom_c
                elif sense == "le" and val > bound:
                    total += t.penalty_k * (val - bound) / denom_c
        return total

    def constraint_funcs_from_terms(
        self,
        evaluator: AbstractEvaluator,
        design_vars: List[Tuple[str, Tuple[float, float]]],
    ) -> List[Callable[[np.ndarray, Dict[str, float]], float]]:
        funcs = []
        for t in self._terms:
            if t.constraint is None:
                continue
            sense, bound = t.constraint
            kpi = t.kpi_key

            def _make(k: str, s: str, b: float):
                def _con(x: np.ndarray, base: Dict[str, float]) -> float:
                    p = dict(base)
                    for i, (n, _) in enumerate(design_vars):
                        p[n] = float(x[i])
                    v = evaluator.evaluate(p).get(k, 0.0)
                    return (v - b) if s == "ge" else (b - v)
                return _con

            funcs.append(_make(kpi, sense, bound))
        return funcs

    @property
    def terms(self) -> List[ObjectiveTerm]:
        return list(self._terms)

class GradientCache:
    """
    Registry for analytical gradient functions.

    Each entry: grad_fn(params, dvar_names) -> np.ndarray of shape (len(dvar_names),)

    Domains register their known analytical gradients here.  The SLSQP wrapper
    queries this cache before falling back to finite differences.
    """

    def __init__(self) -> None:
        self._grad_fns: Dict[str, Callable[[Dict[str, float], List[str]], np.ndarray]] = {}

    def register(self, kpi_key: str,
                 fn: Callable[[Dict[str, float], List[str]], np.ndarray]) -> None:
        self._grad_fns[kpi_key] = fn

    def get(self, kpi_key: str) -> Optional[Callable]:
        return self._grad_fns.get(kpi_key)

    def has(self, kpi_key: str) -> bool:
        return kpi_key in self._grad_fns


# ── Nozzle analytical gradients ───────────────────────────────────────────────

def _nozzle_grad_specific_impulse(params: Dict[str, float],
                                  dvar_names: List[str]) -> np.ndarray:
    """
    Isp = Ve / g0  where Ve = Me * sqrt(gamma*R*T0 / (1 + 0.5*(gamma-1)*Me^2))
    dIsp/dMe and dIsp/dthroat_radius (= 0) — analytical.
    """
    g0    = 9.80665
    gamma = params["gamma"]
    R     = params["R"]
    T0    = params["T0"]
    Me    = params["Me"]
    gm1   = gamma - 1.0
    denom = 1.0 + 0.5 * gm1 * Me**2

    Ve = Me * math.sqrt(gamma * R * T0 / denom)
    # dVe/dMe
    dVe_dMe = (math.sqrt(gamma * R * T0 / denom)
               - Me**2 * 0.5 * gm1 * math.sqrt(gamma * R * T0) / (denom**1.5))
    dIsp_dMe = dVe_dMe / g0
    grad = np.zeros(len(dvar_names))
    for i, n in enumerate(dvar_names):
        if n == "Me":
            grad[i] = dIsp_dMe
        # throat_radius does not appear in Isp
    return grad


def _nozzle_grad_thrust(params: Dict[str, float],
                        dvar_names: List[str]) -> np.ndarray:
    """
    F = m_dot * Ve + (Pe - Pa) * Ae
    Ae = pi * r_t^2 * AR,  AR = A(Me, gamma)
    dF/dMe and dF/dr_t — analytical.
    """
    gamma  = params["gamma"]
    R      = params["R"]
    T0     = params["T0"]
    P0     = params["P0"]
    Pa     = params["Pa"]
    m_dot  = params["m_dot"]
    r_t    = params["throat_radius"]
    Me     = params["Me"]
    gm1    = gamma - 1.0
    gp1    = gamma + 1.0
    denom  = 1.0 + 0.5 * gm1 * Me**2

    Te     = T0 / denom
    Ve     = Me * math.sqrt(gamma * R * Te)
    Pe     = P0 * denom**(-(gamma / gm1))
    AR     = (1.0 / Me) * ((2.0 / gp1) * denom)**(0.5 * gp1 / gm1)
    Ae     = math.pi * r_t**2 * AR

    # dVe/dMe (same as above)
    dVe_dMe = (math.sqrt(gamma * R * T0 / denom)
               - Me**2 * 0.5 * gm1 * math.sqrt(gamma * R * T0) / (denom**1.5))

    # dPe/dMe
    dPe_dMe = P0 * (-(gamma / gm1)) * denom**(-(gamma / gm1) - 1.0) * gm1 * Me

    # dAR/dMe via logarithmic differentiation:
    # AR = (1/Me)*((2/gp1)*(1+0.5*gm1*Me^2))^(0.5*gp1/gm1)
    # ln AR = -ln Me + 0.5*gp1/gm1 * ln((2/gp1)*denom)
    # d(ln AR)/dMe = -1/Me + 0.5*gp1/gm1 * gm1*Me/denom = -1/Me + 0.5*gp1*Me/denom
    dAR_dMe = AR * (-1.0 / Me + 0.5 * gp1 * Me / denom)
    dAe_dMe = math.pi * r_t**2 * dAR_dMe

    dF_dMe = m_dot * dVe_dMe + dPe_dMe * Ae + (Pe - Pa) * dAe_dMe

    # dF/dr_t  (only through Ae = pi*r_t^2*AR)
    dAe_drt  = 2.0 * math.pi * r_t * AR
    dF_drt   = (Pe - Pa) * dAe_drt

    grad = np.zeros(len(dvar_names))
    for i, n in enumerate(dvar_names):
        if n == "Me":
            grad[i] = dF_dMe
        elif n == "throat_radius":
            grad[i] = dF_drt
    return grad


# ── PMSM motor analytical gradients ──────────────────────────────────────────

def _motor_grad_torque(params: Dict[str, float],
                       dvar_names: List[str]) -> np.ndarray:
    """
    T = 1.5 * p * psi * I_q
    psi = B_gap * tau_p * L * kw * N / p
    tau_p = pi * r_r / p
    => T = 1.5 * B_gap * pi * r_r * L * kw * N * I_q / p
    dT/dr_r, dT/dL, dT/dB_gap, dT/dI_q — all linear → trivial.
    """
    p   = params["pole_pairs"]
    N   = params["turns_per_phase"]
    kw  = params.get("winding_factor", 0.933)
    B   = params["B_gap"]
    r_r = params["rotor_radius"]
    L   = params["stack_length"]
    Iq  = params["current_q"]
    k   = 1.5 * math.pi * kw * N / p   # common factor
    # T = k * B * r_r * L * Iq
    grad = np.zeros(len(dvar_names))
    for i, n in enumerate(dvar_names):
        if n == "rotor_radius":
            grad[i] = k * B * L * Iq
        elif n == "stack_length":
            grad[i] = k * B * r_r * Iq
        elif n == "B_gap":
            grad[i] = k * r_r * L * Iq
        elif n == "current_q":
            grad[i] = k * B * r_r * L
    return grad


def _motor_grad_power_density(params: Dict[str, float],
                              dvar_names: List[str]) -> np.ndarray:
    """
    PD = T * omega / (1000 * m)
    m depends on r_r, L (and indirectly geometry constants).
    Use quotient rule: d(T*omega/(1000*m))/dx = omega/(1000*m) * dT/dx
                        - T*omega/(1000*m^2) * dm/dx
    Mass formula (from ElectricMotorGeometryGenerator):
      iron_r  = pi * r_r^2 * L * f_ri * rho_fe
      iron_s  = pi * ((r_r+ag+yoke)^2+extra)^2 - (r_r+ag+yoke)^2) * L * f_si * rho_fe  [approx]
      copper  = slot_area_factor * L * f_cu * rho_cu
      magnets = pi * r_r * L * f_mag_s * f_mag_d * rho_mag
    We provide the gradient of m wrt r_r and L analytically.
    """
    p       = params["pole_pairs"]
    N       = params["turns_per_phase"]
    kw      = params.get("winding_factor", 0.933)
    B       = params["B_gap"]
    r_r     = params["rotor_radius"]
    L       = params["stack_length"]
    Iq      = params["current_q"]
    omega   = params["omega_mech"]

    # Geometry constants (defaults from ElectricMotorConstraints)
    ag      = params.get("airgap_m",                  0.001)
    yoke    = params.get("stator_yoke_thickness_m",   0.015)
    extra   = params.get("stator_outer_extra_m",      0.025)
    saf     = params.get("slot_area_factor_m2_per_m", 0.0008)
    f_ri    = params.get("rotor_iron_vol_frac",       0.60)
    f_si    = params.get("stator_iron_vol_frac",      0.50)
    f_cu    = params.get("copper_fill_factor",        0.40)
    f_ms    = params.get("magnet_surface_frac",       0.15)
    f_md    = params.get("magnet_radial_depth_frac",  0.40)
    hf      = params.get("housing_mass_frac",         0.20)
    rho_fe  = params.get("density_iron_kg_m3",        7650.0)
    rho_cu  = params.get("density_copper_kg_m3",      8960.0)
    rho_mag = params.get("density_magnet_kg_m3",      7500.0)

    r_si  = r_r + ag + yoke
    r_so  = r_si + extra

    # mass components
    vol_ri  = math.pi * r_r**2 * L * f_ri
    vol_si  = math.pi * (r_so**2 - r_si**2) * L * f_si
    vol_mag = math.pi * r_r * L * f_ms * f_md
    vol_cu  = saf * L * f_cu

    m_active = (vol_ri + vol_si) * rho_fe + vol_cu * rho_cu + vol_mag * rho_mag
    m_total  = m_active * (1.0 + hf)

    # Torque and power
    k_T = 1.5 * math.pi * kw * N / p
    T   = k_T * B * r_r * L * Iq
    P   = T * omega

    # dT/dx
    dT = np.zeros(len(dvar_names))
    for i, n in enumerate(dvar_names):
        if n == "rotor_radius":   dT[i] = k_T * B * L * Iq
        elif n == "stack_length": dT[i] = k_T * B * r_r * Iq
        elif n == "B_gap":        dT[i] = k_T * r_r * L * Iq
        elif n == "current_q":    dT[i] = k_T * B * r_r * L

    # dm/dx
    dm = np.zeros(len(dvar_names))
    for i, n in enumerate(dvar_names):
        if n == "rotor_radius":
            dvol_ri  = math.pi * 2 * r_r * L * f_ri
            dvol_si  = math.pi * (2 * r_so * 0 - 2 * r_si * 1) * L * f_si  # dr_si/dr_r = 1
            dvol_mag = math.pi * L * f_ms * f_md
            dm[i] = ((dvol_ri + dvol_si) * rho_fe + dvol_mag * rho_mag) * (1 + hf)
        elif n == "stack_length":
            dvol_ri  = math.pi * r_r**2 * f_ri
            dvol_si  = math.pi * (r_so**2 - r_si**2) * f_si
            dvol_mag = math.pi * r_r * f_ms * f_md
            dvol_cu  = saf * f_cu
            dm[i] = ((dvol_ri + dvol_si) * rho_fe + dvol_cu * rho_cu + dvol_mag * rho_mag) * (1 + hf)

    grad = np.zeros(len(dvar_names))
    if m_total > 1e-9:
        # d(PD)/dx = omega/(1000*m) * dT/dx - T*omega/(1000*m^2) * dm/dx
        grad = (omega / (1000.0 * m_total)) * dT - (P / (1000.0 * m_total**2)) * dm

    return grad

_GRADIENT_CACHE.register("thrust",            _nozzle_grad_thrust)
_GRADIENT_CACHE.register("torque",            _motor_grad_torque)

# §7  GENERIC SLSQP OPTIMIZER  (with analytical-gradient support)
# =============================================================================


class GenericOptimizer(AbstractOptimizer):
    def __init__(self, evaluator: AbstractEvaluator,
                 objective: Callable[[Dict[str, Any]], float],
                 constraint_funcs: List[Callable[[np.ndarray, Dict[str, float]], float]],
                 design_vars: List[Tuple[str, Tuple[float, float]]],
                 use_gradients: bool = True) -> None:
        self._ev    = evaluator
        self._obj   = objective
        self._cons  = constraint_funcs
        self._dvars = design_vars
        self._use_grad = use_gradients
        self._dvar_names = [n for n, _ in design_vars]

    def _unpack(self, x: np.ndarray, base: Dict[str, float]) -> Dict[str, float]:
        p = dict(base)
        for i, (n, _) in enumerate(self._dvars):
            p[n] = float(x[i])
        return p

    def _objective_with_grad(self, x: np.ndarray, base: Dict[str, float]) -> float:
        return self._obj(self._ev.evaluate(self._unpack(x, base)))

    def _build_jac(self, base: Dict[str, float]) -> Optional[Callable]:
        """
        Build a Jacobian callable using analytical gradients where available,
        falling back to central finite differences per parameter otherwise.
        """
        if not self._use_grad:
            return None

        # Check if the evaluator exposes any analytical gradient
        has_any = any(_GRADIENT_CACHE.has(t.kpi_key)
                      for t in (getattr(self._obj, "terms", None) or []))

        if not has_any:
            return None

        def _jac(x: np.ndarray, b: Dict[str, float]) -> np.ndarray:
            p = self._unpack(x, b)
            total_grad = np.zeros(len(self._dvar_names))
            results = self._ev.evaluate(p)
            terms = getattr(self._obj, "terms", [])
            for t in terms:
                grad_fn = _GRADIENT_CACHE.get(t.kpi_key)
                if grad_fn is None:
                    # finite difference fallback for this term
                    eps = 1e-6
                    for i, n in enumerate(self._dvar_names):
                        xp, xm = x.copy(), x.copy()
                        xp[i] += eps; xm[i] -= eps
                        rp = self._ev.evaluate(self._unpack(xp, b)).get(t.kpi_key, 0.0)
                        rm = self._ev.evaluate(self._unpack(xm, b)).get(t.kpi_key, 0.0)
                        nom = abs(t.nominal) if t.nominal != 0.0 else 1.0
                        total_grad[i] += t.weight * (rp - rm) / (2 * eps * nom)
                else:
                    g = grad_fn(p, self._dvar_names)
                    nom = abs(t.nominal) if t.nominal != 0.0 else 1.0
                    total_grad += t.weight * g / nom
            return total_grad

        return _jac

    def optimize_design(self, initial_params: Dict[str, float],
                        x0_override: Optional[np.ndarray] = None) -> OptimResult:
        x0 = x0_override if x0_override is not None else np.array(
            [initial_params.get(n, 0.5*(lo+hi)) for n, (lo, hi) in self._dvars])
        bounds = [b for _, b in self._dvars]
        cons   = [{"type": "ineq", "fun": f, "args": (initial_params,)} for f in self._cons]
        jac    = self._build_jac(initial_params)
        t0 = time.perf_counter()
        options: Dict[str, Any] = {"ftol": 1e-9, "maxiter": 500}
        res = minimize(
            self._objective_with_grad,
            x0, args=(initial_params,), bounds=bounds, constraints=cons,
            method="SLSQP", jac=jac, options=options)
        elapsed    = time.perf_counter() - t0
        opt_params = self._unpack(res.x, initial_params)
        return OptimResult(
            optimized_params=opt_params, results=self._ev.evaluate(opt_params),
            objective_value=float(res.fun), success=bool(res.success),
            message=str(res.message), n_iter=int(res.nit), elapsed_s=elapsed)

# §7b  STRICT FEASIBILITY POST-PROCESSOR + REPAIR STEP
# =============================================================================


class FeasibilityRepair:
    """
    Post-processes an OptimResult:
      1. Checks each constraint g_i(x*, base) >= 0.
      2. If any are violated, runs a repair minimisation:
         minimise  ||x - x*||^2  subject to  g_i(x,base) >= 0  and bounds
         i.e. project x* back to the nearest feasible point.
      3. Flags the result as feasible/infeasible and, if repaired, re-evaluates.
    """

    def __init__(self,
                 constraint_funcs: List[Callable[[np.ndarray, Dict[str, float]], float]],
                 design_vars: List[Tuple[str, Tuple[float, float]]],
                 evaluator: AbstractEvaluator,
                 tol: float = 1e-6) -> None:
        self._cons      = constraint_funcs
        self._dvars     = design_vars
        self._ev        = evaluator
        self._tol       = tol
        self._dvar_names = [n for n, _ in design_vars]

    def _x_from_params(self, params: Dict[str, float]) -> np.ndarray:
        return np.array([params.get(n, 0.0) for n in self._dvar_names])

    def _params_from_x(self, x: np.ndarray, base: Dict[str, float]) -> Dict[str, float]:
        p = dict(base)
        for i, n in enumerate(self._dvar_names):
            p[n] = float(x[i])
        return p

    def _violations(self, x: np.ndarray, base: Dict[str, float]) -> List[float]:
        return [f(x, base) for f in self._cons]

    def is_feasible(self, x: np.ndarray, base: Dict[str, float]) -> bool:
        return all(v >= -self._tol for v in self._violations(x, base))

    def repair(self, result: OptimResult) -> OptimResult:
        base = result.optimized_params
        x_star = self._x_from_params(base)

        if self.is_feasible(x_star, base):
            return OptimResult(**{**result.__dict__, "feasible": True})

        log.info("Feasibility repair: %d constraint(s) violated — projecting to feasible region.",
                 sum(1 for v in self._violations(x_star, base) if v < -self._tol))

        bounds = [b for _, b in self._dvars]
        cons_scipy = [{"type": "ineq", "fun": f, "args": (base,)} for f in self._cons]

        # Projection: minimise distance to x_star subject to constraints + bounds
        def _proj_obj(x: np.ndarray) -> float:
            return float(np.dot(x - x_star, x - x_star))

        def _proj_jac(x: np.ndarray) -> np.ndarray:
            return 2.0 * (x - x_star)

        rep = minimize(
            _proj_obj, x_star, jac=_proj_jac, bounds=bounds,
            constraints=cons_scipy, method="SLSQP",
            options={"ftol": 1e-10, "maxiter": 400})

        x_rep = rep.x
        feasible_after = self.is_feasible(x_rep, base)
        if not feasible_after:
            log.warning("Repair did not achieve full feasibility (residual violations remain).")

        rep_params = self._params_from_x(x_rep, base)
        rep_results = self._ev.evaluate(rep_params)

        # Re-compute objective with repaired params (no penalty needed — genuinely feasible)
        from scipy.optimize import minimize as _min
        rep_obj = float(rep.fun)   # distance-to-x* not the hardware objective; caller should recompute

        return OptimResult(
            optimized_params=rep_params,
            results=rep_results,
            objective_value=result.objective_value,  # report original obj value
            success=result.success and feasible_after,
            message=f"{result.message} | repaired (dist={rep.fun:.4g}, feasible={feasible_after})",
            n_iter=result.n_iter + int(rep.nit),
            elapsed_s=result.elapsed_s,
            feasible=feasible_after,
        )

# §8  DIFFERENTIAL EVOLUTION + SLSQP POLISH OPTIMIZER
# =============================================================================


class DifferentialEvolutionOptimizer(AbstractOptimizer):
    """
    Global search via scipy.optimize.differential_evolution followed by
    an SLSQP polish pass.

    During DE the constraints are penalised with a large exterior coefficient
    (DE has no native inequality-constraint support).  The SLSQP pass then
    enforces them properly.
    """

    def __init__(self, evaluator: AbstractEvaluator,
                 objective: Callable[[Dict[str, Any]], float],
                 constraint_funcs: List[Callable[[np.ndarray, Dict[str, float]], float]],
                 design_vars: List[Tuple[str, Tuple[float, float]]],
                 de_popsize: int = 15, de_maxiter: int = 300,
                 de_tol: float = 1e-7, de_seed: int = 0) -> None:
        self._ev        = evaluator
        self._obj       = objective
        self._cons      = constraint_funcs
        self._dvars     = design_vars
        self._popsize   = de_popsize
        self._maxiter   = de_maxiter
        self._de_tol    = de_tol
        self._de_seed   = de_seed

    def _unpack(self, x: np.ndarray, base: Dict[str, float]) -> Dict[str, float]:
        p = dict(base)
        for i, (n, _) in enumerate(self._dvars):
            p[n] = float(x[i])
        return p

    def _penalised(self, x: np.ndarray, base: Dict[str, float], pen: float = 1e6) -> float:
        obj = self._obj(self._ev.evaluate(self._unpack(x, base)))
        for f in self._cons:
            v = f(x, base)
            if v < 0.0:
                obj += pen * v ** 2
        return obj

    def optimize_design(self, initial_params: Dict[str, float],
                        x0_override: Optional[np.ndarray] = None) -> OptimResult:
        bounds = [b for _, b in self._dvars]
        t0     = time.perf_counter()
        log.info("DE: popsize=%d maxiter=%d seed=%d dims=%d",
                 self._popsize, self._maxiter, self._de_seed, len(self._dvars))

        de_res = differential_evolution(
            self._penalised, bounds, args=(initial_params,),
            popsize=self._popsize, maxiter=self._maxiter, tol=self._de_tol,
            seed=self._de_seed, polish=False,
            mutation=(0.5, 1.0), recombination=0.7, workers=1)
        log.info("DE done: obj=%.4g  nfev=%d", de_res.fun, de_res.nfev)

        cons_scipy = [{"type": "ineq", "fun": f, "args": (initial_params,)}
                      for f in self._cons]
        pol = minimize(
            lambda x, b: self._obj(self._ev.evaluate(self._unpack(x, b))),
            de_res.x, args=(initial_params,), bounds=bounds,
            constraints=cons_scipy, method="SLSQP",
            options={"ftol": 1e-10, "maxiter": 500})

        elapsed    = time.perf_counter() - t0
        best_x     = pol.x if pol.fun <= de_res.fun else de_res.x
        opt_params = self._unpack(best_x, initial_params)
        return OptimResult(
            optimized_params=opt_params, results=self._ev.evaluate(opt_params),
            objective_value=float(min(pol.fun, de_res.fun)),
            success=bool(pol.success or de_res.success),
            message=f"DE nfev={de_res.nfev}; polish: {pol.message}",
            n_iter=int(de_res.nit + pol.nit), elapsed_s=elapsed)

def _evaluate_one(args: Tuple[Dict[str, float], str, str]) -> Tuple[int, Dict[str, Any]]:
    """
    Top-level function (required for pickle) used by ProcessPoolExecutor.
    args = (params_dict, domain_name, config_json_str)
    Returns (index, result_dict).
    """
    idx, params, domain_name, config_json = args
    config = DomainConfig(**json.loads(config_json))
    domain = get_domain(domain_name, config)
    result = domain.evaluator.evaluate(params)
    # Strip non-serialisable geometry dict
    result_clean = {k: v for k, v in result.items()
                    if isinstance(v, (int, float, str, bool))}
    return idx, result_clean


class ParallelEvaluator:
    """
    Wraps a list of parameter dicts and evaluates them concurrently using
    ProcessPoolExecutor.  Typical speedup: 4–20× on a laptop with 4–8 cores.

    Usage:
        pe = ParallelEvaluator(domain_name, config, max_workers=6)
        results = pe.evaluate_batch(list_of_param_dicts)
    """

    def __init__(self, domain_name: str, config: DomainConfig,
                 max_workers: Optional[int] = None) -> None:
        self._domain_name = domain_name
        self._config_json = json.dumps({
            "params":      config.params,
            "bounds":      {k: list(v) for k, v in config.bounds.items()},
            "weights":     config.weights,
            "tols":        config.tols,
            "mass_models": config.mass_models,
        })
        self._max_workers = max_workers

    def evaluate_batch(self, param_list: List[Dict[str, float]]) -> List[Dict[str, Any]]:
        """Evaluate a batch of parameter dicts in parallel, preserving order."""
        tasks = [(i, p, self._domain_name, self._config_json)
                 for i, p in enumerate(param_list)]
        results: Dict[int, Dict[str, Any]] = {}
        with ProcessPoolExecutor(max_workers=self._max_workers) as pool:
            futures = {pool.submit(_evaluate_one, t): t[0] for t in tasks}
            for fut in as_completed(futures):
                try:
                    idx, res = fut.result()
                    results[idx] = res
                except Exception as exc:
                    idx = futures[fut]
                    log.warning("Parallel eval failed for index %d: %s", idx, exc)
                    results[idx] = {}
        return [results[i] for i in range(len(param_list))]


class ParallelMultiStartOptimizer:
    """
    Runs multi-start SLSQP concurrently.  Each worker runs a full
    GenericOptimizer.optimize_design() call from a different starting point.
    Falls back to sequential if max_workers=1.
    """

    def __init__(self, domain_name: str, config: DomainConfig,
                 design_vars: List[Tuple[str, Tuple[float, float]]],
                 max_workers: Optional[int] = None) -> None:
        self._domain_name = domain_name
        self._config      = config
        self._dvars       = design_vars
        self._max_workers = max_workers

    def run(self, start_points: List[np.ndarray],
            base_params: Dict[str, float]) -> "MultiStartResults":
        """Run all start points in parallel, return aggregated MultiStartResults."""
        t_total = time.perf_counter()
        all_runs: List[OptimResult] = [None] * len(start_points)  # type: ignore

        def _run_one(args):
            idx, x0 = args
            domain  = get_domain(self._domain_name, self._config)
            opt     = domain.optimizer
            result  = opt.optimize_design(base_params, x0_override=x0)
            return idx, result

        if self._max_workers == 1:
            for idx, x0 in enumerate(start_points):
                _, res = _run_one((idx, x0))
                all_runs[idx] = res
        else:
            # Use threads here (not processes) since optimizer objects aren't pickable
            from concurrent.futures import ThreadPoolExecutor
            with ThreadPoolExecutor(max_workers=self._max_workers) as pool:
                futs = {pool.submit(_run_one, (i, x0)): i
                        for i, x0 in enumerate(start_points)}
                for fut in as_completed(futs):
                    try:
                        idx, res = fut.result()
                        all_runs[idx] = res
                    except Exception as exc:
                        log.warning("Parallel start %d failed: %s", futs[fut], exc)

        elapsed   = time.perf_counter() - t_total
        converged = [r for r in all_runs if r is not None and r.success]
        best      = min(converged or [r for r in all_runs if r is not None],
                        key=lambda r: r.objective_value)
        return MultiStartResults(
            all_runs=[r for r in all_runs if r is not None],
            best=best,
            n_converged=len(converged),
            mode="parallel",
            n_runs=len(start_points),
            seed=0,
            total_elapsed_s=elapsed,
        )


# =============================================================================
# §9  SAMPLING ENGINE  (LHS / Sobol multi-start)
# =============================================================================

SamplingMode = str


def _lhs(n: int, d: int, seed: int) -> np.ndarray:
    return qmc.LatinHypercube(d=d, seed=seed).random(n=n)

def _sobol(n: int, d: int, seed: int) -> np.ndarray:
    m = math.ceil(math.log2(max(n, 2)))
    return qmc.Sobol(d=d, scramble=True, seed=seed).random_base2(m=m)[:n]

def _scale(unit: np.ndarray, bounds: List[Tuple[float, float]]) -> np.ndarray:
    lo = np.array([b[0] for b in bounds])
    hi = np.array([b[1] for b in bounds])
    return lo + unit * (hi - lo)


@dataclass
class MultiStartResults:
    all_runs:       List[OptimResult]
    best:           OptimResult
    n_converged:    int
    mode:           SamplingMode
    n_runs:         int
    seed:           int
    total_elapsed_s: float

    def summary_table(self) -> str:
        hdr = f"{'Run':>4}  {'Conv':>4}  {'Feas':>4}  {'Iters':>5}  {'Objective':>14}  {'Time(s)':>9}"
        sep = "-" * len(hdr)
        rows = [hdr, sep]
        for i, r in enumerate(self.all_runs):
            feas = getattr(r, "feasible", True)
            rows.append(f"{i+1:>4}  {'YES' if r.success else 'NO ':>4}  "
                        f"{'YES' if feas else 'NO ':>4}  {r.n_iter:>5}  "
                        f"{r.objective_value:>14.6g}  {r.elapsed_s:>9.3f}")
        rows += [sep, f"Best obj={self.best.objective_value:.6g}  |  "
                      f"Converged: {self.n_converged}/{self.n_runs}  |  "
                      f"Total: {self.total_elapsed_s:.2f}s"]
        return "\n".join(rows)


def run_multistart(optimizer: AbstractOptimizer, base_params: Dict[str, float],
                   design_vars: List[Tuple[str, Tuple[float, float]]],
                   mode: SamplingMode, n_runs: int, seed: int,
                   feasibility_repair: Optional[FeasibilityRepair] = None,
                   max_workers: int = 1) -> MultiStartResults:
    d, bounds = len(design_vars), [b for _, b in design_vars]
    pts = _scale((_lhs if mode == "monte_carlo" else _sobol)(n_runs, d, seed), bounds)
    log.info("Multi-start [%s]: %d runs, seed=%d, %d vars, workers=%d",
             mode, n_runs, seed, d, max_workers)
    all_runs: List[OptimResult] = []
    t0 = time.perf_counter()

    if max_workers > 1:
        from concurrent.futures import ThreadPoolExecutor
        def _one(args):
            i, x0 = args
            log.debug("  run %d/%d  x0=%s", i+1, n_runs, np.round(x0, 4))
            r = optimizer.optimize_design(base_params, x0_override=x0)
            if feasibility_repair is not None:
                r = feasibility_repair.repair(r)
            return i, r
        results_map: Dict[int, OptimResult] = {}
        with ThreadPoolExecutor(max_workers=max_workers) as pool:
            futs = {pool.submit(_one, (i, x0)): i for i, x0 in enumerate(pts)}
            for fut in as_completed(futs):
                try:
                    i, r = fut.result()
                    results_map[i] = r
                except Exception as exc:
                    log.warning("Run %d failed: %s", futs[fut], exc)
        all_runs = [results_map[i] for i in range(n_runs) if i in results_map]
    else:
        for i, x0 in enumerate(pts):
            log.debug("  run %d/%d  x0=%s", i+1, n_runs, np.round(x0, 4))
            r = optimizer.optimize_design(base_params, x0_override=x0)
            if feasibility_repair is not None:
                r = feasibility_repair.repair(r)
            all_runs.append(r)

    total = time.perf_counter() - t0
    converged = [r for r in all_runs if r.success]
    feasible  = [r for r in all_runs if getattr(r, "feasible", True)]
    best = min((feasible or converged or all_runs), key=lambda r: r.objective_value)
    return MultiStartResults(all_runs=all_runs, best=best, n_converged=len(converged),
                             mode=mode, n_runs=n_runs, seed=seed, total_elapsed_s=total)

@dataclass
class ToleranceMCResults:
    n_samples:    int;   pass_rate: float
    kpi_mean:     Dict[str, float];  kpi_std:  Dict[str, float]
    kpi_p05:      Dict[str, float];  kpi_p95:  Dict[str, float]
    tols_used:    Dict[str, float];  elapsed_s: float

    def report(self) -> str:
        lines = [
            f"  Tolerance Monte-Carlo  ({self.n_samples} samples)",
            f"  Pass rate            = {self.pass_rate * 100:.1f} %",
            f"  Tolerance σ (frac.)  = "
            + str({k: f"{v:.4f}" for k, v in self.tols_used.items()}),
            "",
            f"  {'KPI':<30}  {'Mean':>12}  {'Std':>10}  {'P05':>12}  {'P95':>12}",
            "  " + "-" * 70,
        ]
        for k in self.kpi_mean:
            lines.append(f"  {k:<30}  {self.kpi_mean[k]:>12.5g}  {self.kpi_std[k]:>10.4g}"
                         f"  {self.kpi_p05[k]:>12.5g}  {self.kpi_p95[k]:>12.5g}")
        lines.append(f"\n  Elapsed: {self.elapsed_s:.2f}s")
        return "\n".join(lines)

    def as_dict(self) -> Dict[str, Any]:
        """Serialise to a flat dict suitable for Streamlit display."""
        rows = []
        for k in self.kpi_mean:
            rows.append({
                "KPI": k,
                "Mean": self.kpi_mean[k],
                "Std": self.kpi_std[k],
                "P05": self.kpi_p05[k],
                "P95": self.kpi_p95[k],
            })
        return {"pass_rate": self.pass_rate, "n_samples": self.n_samples, "kpis": rows}


def run_tolerance_mc(
    best_params: Dict[str, float], evaluator: AbstractEvaluator,
    constraint_funcs: List[Callable[[np.ndarray, Dict[str, float]], float]],
    design_vars: List[Tuple[str, Tuple[float, float]]],
    tols: Dict[str, float], n_samples: int, seed: int,
) -> ToleranceMCResults:
    rng        = np.random.default_rng(seed)
    tol_names  = list(tols.keys())
    tol_sigmas = np.array([abs(best_params.get(k, 1.0)) * tols[k] for k in tol_names])
    dvar_names = [n for n, _ in design_vars]
    x_best     = np.array([best_params.get(n, 0.0) for n in dvar_names])
    perturbs   = rng.normal(0.0, tol_sigmas, size=(n_samples, len(tol_names)))

    # Clamp perturbed values to design-var bounds to prevent numerical blow-up
    dvar_bound_map = dict(design_vars)

    all_scalar: List[Dict[str, float]] = []
    n_pass = 0
    t0 = time.perf_counter()

    for delta in perturbs:
        p = dict(best_params)
        for i, name in enumerate(tol_names):
            raw_val = best_params.get(name, 0.0) + delta[i]
            if name in dvar_bound_map:
                lo, hi = dvar_bound_map[name]
                raw_val = max(lo, min(hi, raw_val))
            p[name] = raw_val
        try:
            result   = evaluator.evaluate(p)
            feasible = all(f(x_best, p) >= -1e-6 for f in constraint_funcs)
            if feasible:
                n_pass += 1
            all_scalar.append({k: float(v) for k, v in result.items()
                                if isinstance(v, (int, float)) and k != "geometry"})
        except Exception:
            pass

    elapsed = time.perf_counter() - t0
    if not all_scalar:
        raise RuntimeError("Tolerance MC: no valid samples produced.")

    keys = list(all_scalar[0].keys())
    arr  = np.array([[s.get(k, float("nan")) for k in keys] for s in all_scalar])
    return ToleranceMCResults(
        n_samples=n_samples, pass_rate=n_pass / max(len(all_scalar), 1),
        kpi_mean={k: float(np.nanmean(arr[:, i]))           for i, k in enumerate(keys)},
        kpi_std= {k: float(np.nanstd(arr[:, i]))            for i, k in enumerate(keys)},
        kpi_p05= {k: float(np.nanpercentile(arr[:, i],  5)) for i, k in enumerate(keys)},
        kpi_p95= {k: float(np.nanpercentile(arr[:, i], 95)) for i, k in enumerate(keys)},
        tols_used=dict(tols), elapsed_s=elapsed)

class MassModelRegistry:
    """
    Allows mass calculations to be overridden via JSON tables or small empirical
    regressions loaded from DomainConfig.mass_models.

    Supported model_type values:
      "table"       – piecewise-linear interpolation from a JSON table
                      { "x_key": "param_name", "x": [...], "mass_kg": [...] }
      "regression"  – polynomial regression
                      { "coeffs": [a0, a1, a2, ...], "input_keys": ["k1","k2",...] }
                      mass = a0 + a1*k1 + a2*k2 + ...  (linear in inputs)
      "constant"    – fixed mass override  { "value_kg": 12.5 }
    """

    def __init__(self, domain_name: str, mass_models_cfg: Dict[str, Any]) -> None:
        self._domain = domain_name
        self._cfg    = mass_models_cfg.get(domain_name, {})

    def has_override(self) -> bool:
        return bool(self._cfg)

    def calculate(self, params: Dict[str, float],
                  fallback_fn: Callable[[Dict[str, float]], float]) -> float:
        if not self.has_override():
            return fallback_fn(params)

        model_type = self._cfg.get("model_type", "constant")

        if model_type == "constant":
            return float(self._cfg["value_kg"])

        elif model_type == "table":
            x_key  = self._cfg["x_key"]
            xs     = np.array(self._cfg["x"], dtype=float)
            masses = np.array(self._cfg["mass_kg"], dtype=float)
            x_val  = params.get(x_key, xs[0])
            return float(np.interp(x_val, xs, masses))

        elif model_type == "regression":
            coeffs  = self._cfg["coeffs"]
            in_keys = self._cfg["input_keys"]
            values  = [params.get(k, 0.0) for k in in_keys]
            return float(sum(c * v for c, v in zip(coeffs[1:], values)) + coeffs[0])

        else:
            log.warning("MassModelRegistry: unknown model_type '%s', using fallback.", model_type)
            return fallback_fn(params)

def _nozzle_exit_mach(subs: Dict[str, float]) -> float:
    if "area_ratio" not in subs:
        return subs["Me"]
    gamma, ar = subs["gamma"], subs["area_ratio"]
    gp1, gm1  = gamma + 1.0, gamma - 1.0
    def _f(M: float) -> float:
        return ((2.0/gp1)*(1.0+0.5*gm1*M*M))**(0.5*gp1/gm1)/M - ar
    return brentq(_f, 1.001, 20.0)

def _nozzle_exit_velocity(subs: Dict[str, float]) -> float:
    gamma, R, T0, Me = subs["gamma"], subs["R"], subs["T0"], subs["Me"]
    Te = T0 / (1.0 + 0.5*(gamma-1.0)*Me**2)
    return Me * math.sqrt(gamma * R * Te)

def _nozzle_thrust(subs: Dict[str, float]) -> float:
    gamma, P0, Pa, Me = subs["gamma"], subs["P0"], subs["Pa"], subs["Me"]
    Pe = P0 * (1.0+0.5*(gamma-1.0)*Me**2)**(-(gamma/(gamma-1.0)))
    Ae = math.pi * subs["throat_radius"]**2 * subs["area_ratio"]
    return subs["m_dot"] * subs["exit_velocity"] + (Pe - Pa) * Ae

def _nozzle_specific_impulse(subs: Dict[str, float]) -> float:
    return subs["thrust"] / (subs["m_dot"] * 9.80665)

def _build_nozzle_registry() -> PhysicsRegistry:
    reg = PhysicsRegistry("nozzle")
    reg.register({"exit_mach": _nozzle_exit_mach, "exit_velocity": _nozzle_exit_velocity,
                  "thrust": _nozzle_thrust, "specific_impulse": _nozzle_specific_impulse})
    return reg


class NozzleAxioms(AbstractAxioms):
    def __init__(self) -> None: self._reg = _build_nozzle_registry()
    def evaluate(self, q: str, s: Dict[str, float]) -> float: return self._reg.evaluate(q, s)


@dataclass
class NozzleConstraints(AbstractConstraints):

    min_specific_impulse_s: float = 200.0;   material_density_kg_m3: float = 8900.0
    wall_thickness_m: float = 0.005;         half_cone_angle_deg: float = 15.0

    def get_funcs(self) -> List[Callable[[np.ndarray, Dict[str, float]], float]]:
        # Provide the canonical constraint function list for use by FeasibilityRepair.
        # (populated properly by WeightedObjective.constraint_funcs_from_terms at domain build)
        return []


class NozzleGeometryGenerator(AbstractGeometryGenerator):
    def __init__(self, c: NozzleConstraints) -> None: self._c = c

    def generate_geometry(self, params: Dict[str, float]) -> Dict[str, float]:
        r_t, Me, gamma = params["throat_radius"], params["Me"], params["gamma"]
        gp1, gm1   = gamma+1.0, gamma-1.0
        ar = (1.0/Me)*((2.0/gp1)*(1.0+0.5*gm1*Me**2))**(0.5*gp1/gm1)
        r_e    = r_t * math.sqrt(ar)
        length = (r_e-r_t) / math.tan(math.radians(self._c.half_cone_angle_deg))
        return {"throat_radius": r_t, "exit_radius": r_e, "area_ratio": ar, "nozzle_length": length}

    def calculate_mass(self, geo: Dict[str, float], params: Dict[str, float]) -> float:
        r_t, r_e, L = geo["throat_radius"], geo["exit_radius"], geo["nozzle_length"]
        slant = math.sqrt(L**2 + (r_e-r_t)**2)
        return math.pi*(r_t+r_e)*slant * self._c.wall_thickness_m * self._c.material_density_kg_m3

    def export_geometry(self, geo: Dict[str, float]) -> str:
        return (f"Nozzle:\n  Throat r={geo['throat_radius']:.4f} m  "
                f"Exit r={geo['exit_radius']:.4f} m  "
                f"AR={geo['area_ratio']:.3f}  L={geo['nozzle_length']:.4f} m")

    def __init__(self, ax: NozzleAxioms) -> None: self._ax = ax
    def simulate(self, params: Dict[str, float]) -> Dict[str, float]:
        s = dict(params)
        s["exit_mach"]        = self._ax.evaluate("exit_mach", s)
        s["Me"]               = s["exit_mach"]
        s["exit_velocity"]    = self._ax.evaluate("exit_velocity", s)
        s["thrust"]           = self._ax.evaluate("thrust", s)
        s["specific_impulse"] = self._ax.evaluate("specific_impulse", s)
        return s


class NozzleEvaluator(AbstractEvaluator):
    def __init__(self, ax: NozzleAxioms, gen: NozzleGeometryGenerator,
                 mass_registry: Optional[MassModelRegistry] = None) -> None:
        self._ax, self._gen = ax, gen
        self._sim = NozzleSimulator(ax)
        self._mass_reg = mass_registry

    @property
    def tolerance_params(self) -> List[str]: return ["throat_radius", "T0", "P0"]

    def evaluate(self, params: Dict[str, float]) -> Dict[str, Any]:
        p = dict(params)
        geo = self._gen.generate_geometry(p)
        p["area_ratio"] = geo["area_ratio"]
        if self._mass_reg and self._mass_reg.has_override():
            mass = self._mass_reg.calculate(p, lambda pp: self._gen.calculate_mass(geo, pp))
        else:
            mass = self._gen.calculate_mass(geo, p)
        sim  = self._sim.simulate(p)
        typed = NozzleResults(exit_mach=sim["exit_mach"], exit_velocity=sim["exit_velocity"],
                              thrust=sim["thrust"], specific_impulse=sim["specific_impulse"],
                              area_ratio=geo["area_ratio"], mass=mass,
                              throat_radius=geo["throat_radius"], exit_radius=geo["exit_radius"],
                              nozzle_length=geo["nozzle_length"])
        out = typed.scalar_metrics(); out["geometry"] = geo; return out

    def gradient(self, quantity: str, params: Dict[str, float],
                 design_var_names: List[str]) -> Optional[np.ndarray]:
        fn = _GRADIENT_CACHE.get(quantity)
        if fn is not None:
            return fn(params, design_var_names)
        return None

def _pmsm_flux_linkage(subs: Dict[str, float]) -> float:
    p = subs["pole_pairs"]
    tau = math.pi * subs["rotor_radius"] / p
    return subs["B_gap"]*tau*subs["stack_length"]*subs.get("winding_factor",0.93)*subs["turns_per_phase"]/p

def _pmsm_torque(subs: Dict[str, float]) -> float:
    psi = subs["flux_linkage_pm"] if "flux_linkage_pm" in subs else _pmsm_flux_linkage(subs)
    return 1.5 * subs["pole_pairs"] * psi * subs["current_q"]

def _pmsm_power(subs: Dict[str, float]) -> float:
    T = subs["torque"] if "torque" in subs else _pmsm_torque(subs)
    return T * subs["omega_mech"]

def _pmsm_power_density(subs: Dict[str, float]) -> float:
    P_kW = (subs["power"] if "power" in subs else _pmsm_power(subs)) / 1000.0
    m    = subs["mass_total"]
    return P_kW/m if m > 1e-9 else 0.0

def _build_motor_registry() -> PhysicsRegistry:
    reg = PhysicsRegistry("electric_motor")
    reg.register({"flux_linkage_pm": _pmsm_flux_linkage, "torque": _pmsm_torque,
                  "power": _pmsm_power, "power_density": _pmsm_power_density})
    return reg

    def evaluate(self, q: str, s: Dict[str, float]) -> float: return self._reg.evaluate(q, s)

    stator_outer_extra_m: float = 0.025;       slot_area_factor_m2_per_m: float = 0.0008

class ElectricMotorGeometryGenerator(AbstractGeometryGenerator):
    def __init__(self, c: ElectricMotorConstraints) -> None: self._c = c

    def generate_geometry(self, params: Dict[str, float]) -> Dict[str, float]:
        r_r, L, c = params["rotor_radius"], params["stack_length"], self._c
        r_si = r_r + c.airgap_m + c.stator_yoke_thickness_m
        r_so = r_si + c.stator_outer_extra_m
        return {"r_rotor": r_r, "r_stator_inner": r_si, "r_stator_outer": r_so,
                "stack_length": L, "slot_area": c.slot_area_factor_m2_per_m*L}

    def calculate_mass(self, geo: Dict[str, float], params: Dict[str, float]) -> float:
        c = self._c
        r_r, r_si, r_so, L = geo["r_rotor"], geo["r_stator_inner"], geo["r_stator_outer"], geo["stack_length"]
        vol_ri  = math.pi*r_r**2*L*c.rotor_iron_vol_frac
        vol_si  = math.pi*(r_so**2-r_si**2)*L*c.stator_iron_vol_frac
        vol_mag = math.pi*r_r*L*c.magnet_surface_frac*c.magnet_radial_depth_frac
        vol_cu  = geo["slot_area"]*c.copper_fill_factor
        m_active = ((vol_ri+vol_si)*c.density_iron_kg_m3 + vol_cu*c.density_copper_kg_m3
                    + vol_mag*c.density_magnet_kg_m3)
        return m_active*(1.0+c.housing_mass_frac)

    def export_geometry(self, geo: Dict[str, float]) -> str:
        return (f"PMSM:\n  Rotor r={geo['r_rotor']:.4f} m  "
                f"Stator outer r={geo['r_stator_outer']:.4f} m  "
                f"L={geo['stack_length']:.4f} m  "
                f"Slot area={geo['slot_area']*1e6:.1f} mm²")

# §13  CENTRIFUGAL PUMP DOMAIN
# =============================================================================
# Euler turbomachinery + Cordier efficiency + Stodola slip factor
# Head:   H = σ·u2·(u2 - V_r2/tan β2)/g
# Power:  P = ρ·g·Q·H / η_h
# Ns_dim: ω√Q / (gH)^0.75  (dimensionless)

def _pump_tip_speed(subs: Dict[str, float]) -> float:
    return math.pi * subs["rotational_speed_rpm"] * subs["impeller_diameter_m"] / 60.0

def _pump_slip_factor(subs: Dict[str, float]) -> float:
    beta2 = math.radians(subs["blade_angle_deg"])
    return 1.0 - math.pi * math.sin(beta2) / subs["n_blades"]

def _pump_radial_velocity(subs: Dict[str, float]) -> float:
    D2 = subs["impeller_diameter_m"]
    b2 = subs["blade_width_ratio"] * D2
    return subs["flow_rate_m3s"] / (math.pi * D2 * b2)

def _pump_head(subs: Dict[str, float]) -> float:
    g     = 9.80665
    u2    = subs.get("tip_speed",       _pump_tip_speed(subs))
    sigma = subs.get("slip_factor",     _pump_slip_factor(subs))
    V_r2  = subs.get("radial_velocity", _pump_radial_velocity(subs))
    tan_b = max(abs(math.tan(math.radians(subs["blade_angle_deg"]))), 1e-9)
    return sigma * u2 * (u2 - V_r2 / tan_b) / g

def _pump_specific_speed(subs: Dict[str, float]) -> float:
    g     = 9.80665
    H     = subs.get("head", _pump_head(subs))
    if H <= 0: return 0.0
    omega = subs["rotational_speed_rpm"] * math.pi / 30.0
    return omega * math.sqrt(subs["flow_rate_m3s"]) / (g * H) ** 0.75

def _pump_hydraulic_efficiency(subs: Dict[str, float]) -> float:
    Ns  = subs.get("specific_speed", _pump_specific_speed(subs))
    eta = 0.65 + 0.30 * math.exp(-0.8 * (Ns - 0.6) ** 2)
    return max(0.50, min(0.92, eta))

def _pump_shaft_power(subs: Dict[str, float]) -> float:
    g   = 9.80665
    H   = subs.get("head", _pump_head(subs))
    eta = subs.get("hydraulic_efficiency", _pump_hydraulic_efficiency(subs))
    return subs["fluid_density_kgm3"] * g * subs["flow_rate_m3s"] * H / max(eta, 1e-6)

def _pump_npsh_r(subs: Dict[str, float]) -> float:
    g     = 9.80665
    u2    = subs.get("tip_speed", _pump_tip_speed(subs))
    r2    = subs["impeller_diameter_m"] / 2.0
    r1    = r2 * subs["inlet_diameter_ratio"]
    omega = subs["rotational_speed_rpm"] * math.pi / 30.0
    u1    = omega * r1
    return (u2**2 / (2*g)) * (1 + (u1/u2)**2) * 0.1

def _build_pump_registry() -> PhysicsRegistry:
    reg = PhysicsRegistry("centrifugal_pump")
    reg.register({"tip_speed": _pump_tip_speed, "slip_factor": _pump_slip_factor,
                  "radial_velocity": _pump_radial_velocity, "head": _pump_head,
                  "specific_speed": _pump_specific_speed,
                  "hydraulic_efficiency": _pump_hydraulic_efficiency,
                  "shaft_power": _pump_shaft_power, "npsh_r": _pump_npsh_r})
    return reg


class PumpAxioms(AbstractAxioms):
    def __init__(self) -> None: self._reg = _build_pump_registry()
    def evaluate(self, q: str, s: Dict[str, float]) -> float: return self._reg.evaluate(q, s)


@dataclass
class PumpConstraints(AbstractConstraints):
    min_head_m: float = 40.0;              max_npsh_r_m: float = 8.0
    max_shaft_power_W: float = 80_000.0;   min_efficiency: float = 0.55
    max_tip_speed_m_s: float = 40.0;       impeller_density_kg_m3: float = 7800.0
    impeller_thickness_m: float = 0.008

    def get_funcs(self) -> List[Callable[[np.ndarray, Dict[str, float]], float]]:
        return []


class PumpGeometryGenerator(AbstractGeometryGenerator):
    def __init__(self, c: PumpConstraints) -> None: self._c = c

    def generate_geometry(self, params: Dict[str, float]) -> Dict[str, float]:
        D2 = params["impeller_diameter_m"]
        return {"impeller_diameter": D2, "blade_width": params["blade_width_ratio"]*D2,
                "inlet_diameter": params["inlet_diameter_ratio"]*D2,
                "blade_angle_deg": params["blade_angle_deg"]}

    def calculate_mass(self, geo: Dict[str, float], params: Dict[str, float]) -> float:
        D2, b2, t, rho = geo["impeller_diameter"], geo["blade_width"], \
                         self._c.impeller_thickness_m, self._c.impeller_density_kg_m3
        disk_vol  = math.pi * (D2/2)**2 * t
        blade_vol = params["n_blades"] * D2 * b2 * t * 0.5
        return (disk_vol + blade_vol) * rho

    def export_geometry(self, geo: Dict[str, float]) -> str:
        return (f"Pump impeller:\n  D2={geo['impeller_diameter']*1000:.1f} mm  "
                f"b2={geo['blade_width']*1000:.1f} mm  "
                f"D1={geo['inlet_diameter']*1000:.1f} mm  "
                f"β2={geo['blade_angle_deg']:.1f}°")


class PumpSimulator(AbstractSimulator):
    def __init__(self, ax: PumpAxioms) -> None: self._ax = ax
    def simulate(self, params: Dict[str, float]) -> Dict[str, float]:
        s = dict(params)
        s["tip_speed"]            = self._ax.evaluate("tip_speed", s)
        s["slip_factor"]          = self._ax.evaluate("slip_factor", s)
        s["radial_velocity"]      = self._ax.evaluate("radial_velocity", s)
        s["head"]                 = self._ax.evaluate("head", s)
        s["specific_speed"]       = self._ax.evaluate("specific_speed", s)
        s["hydraulic_efficiency"] = self._ax.evaluate("hydraulic_efficiency", s)
        s["shaft_power"]          = self._ax.evaluate("shaft_power", s)
        s["npsh_r"]               = self._ax.evaluate("npsh_r", s)
        return s


class PumpEvaluator(AbstractEvaluator):
    def __init__(self, ax: PumpAxioms, gen: PumpGeometryGenerator, c: PumpConstraints,
                 mass_registry: Optional[MassModelRegistry] = None) -> None:
        self._ax, self._gen, self._c = ax, gen, c
        self._sim = PumpSimulator(ax)
        self._mass_reg = mass_registry

    @property
    def tolerance_params(self) -> List[str]:
        return ["impeller_diameter_m", "blade_angle_deg", "rotational_speed_rpm"]

    def evaluate(self, params: Dict[str, float]) -> Dict[str, Any]:
        p = dict(params)
        geo  = self._gen.generate_geometry(p)
        if self._mass_reg and self._mass_reg.has_override():
            mass = self._mass_reg.calculate(p, lambda pp: self._gen.calculate_mass(geo, pp))
        else:
            mass = self._gen.calculate_mass(geo, p)
        sim  = self._sim.simulate(p)
        typed = PumpResults(head_m=sim["head"], shaft_power_W=sim["shaft_power"],
                            hydraulic_efficiency=sim["hydraulic_efficiency"],
                            specific_speed=sim["specific_speed"], tip_speed_m_s=sim["tip_speed"],
                            mass_impeller_kg=mass,
                            power_per_kg_W_kg=sim["shaft_power"]/max(mass,1e-9),
                            npsh_r_m=sim["npsh_r"])
        out = typed.scalar_metrics(); out["geometry"] = geo; return out


class CentrifugalPumpDomain(HardwareDomain):
    def _cache(self, name: str, factory: Callable) -> Any:
        if not hasattr(self, name): object.__setattr__(self, name, factory())
        return getattr(self, name)

    @property
    def axioms(self)      -> PumpAxioms:
        return self._cache("_ax",  PumpAxioms)
    @property
    def constraints(self) -> PumpConstraints:
        return self._cache("_con", PumpConstraints)
    @property
    def geometry_generator(self) -> PumpGeometryGenerator:
        return self._cache("_gen", lambda: PumpGeometryGenerator(self.constraints))
    @property
    def evaluator(self) -> PumpEvaluator:
        mass_reg = MassModelRegistry("centrifugal_pump", self._config.mass_models)
        return self._cache("_ev", lambda: PumpEvaluator(
            self.axioms, self.geometry_generator, self.constraints, mass_reg))

    @property
    def default_weights(self) -> Dict[str, float]:
        return {"shaft_power_W": 1.0, "hydraulic_efficiency": -2.0,
                "head_m": -0.3, "npsh_r_m": 0.1}

    def _build_objective(self) -> WeightedObjective:
        c = self.constraints
        return WeightedObjective([
            ObjectiveTerm("shaft_power_W",        1.0, 30_000.0,  ("le", c.max_shaft_power_W), 100.0),
            ObjectiveTerm("hydraulic_efficiency", -2.0, 0.75,     ("ge", c.min_efficiency),    200.0),
            ObjectiveTerm("head_m",               -0.3, c.min_head_m, ("ge", c.min_head_m),    200.0),
            ObjectiveTerm("npsh_r_m",              0.1, 4.0,      ("le", c.max_npsh_r_m),       50.0),
            ObjectiveTerm("tip_speed_m_s",         0.0, c.max_tip_speed_m_s,
                          ("le", c.max_tip_speed_m_s), 100.0),
        ], self.effective_weights())

    @property
    def optimizer(self) -> GenericOptimizer:
        wo = self._build_objective()
        return GenericOptimizer(self.evaluator, wo,
                                wo.constraint_funcs_from_terms(self.evaluator, self.effective_design_vars()),
                                self.effective_design_vars())

    @property
    def design_vars(self) -> List[Tuple[str, Tuple[float, float]]]:
        return [("impeller_diameter_m", (0.15, 0.50)),
                ("blade_angle_deg",     (15.0, 45.0)),
                ("rotational_speed_rpm",(800.0, 3000.0))]

    @property
    def constraint_funcs(self) -> List[Callable[[np.ndarray, Dict[str, float]], float]]:
        return self._build_objective().constraint_funcs_from_terms(
            self.evaluator, self.effective_design_vars())

    @property
    def default_params(self) -> Dict[str, float]: return PumpParams().to_dict()
    def default_tols(self) -> Dict[str, float]:
        return {"impeller_diameter_m": 0.003, "blade_angle_deg": 0.02, "rotational_speed_rpm": 0.005}

_DOMAIN_MAP: Dict[str, Type[HardwareDomain]] = {
    "nozzle":           NozzleDomain,

def get_domain(name: str, config: DomainConfig) -> HardwareDomain:
    if name not in _DOMAIN_MAP:
        raise ValueError(f"Domain '{name}' not supported. Available: {sorted(_DOMAIN_MAP)}")
    return _DOMAIN_MAP[name](config)


def get_optimizer(domain_obj: HardwareDomain, mode: SamplingMode,
                  de_seed: int = 0) -> AbstractOptimizer:
    if mode == "differential_evo":
        wo    = domain_obj._build_objective()
        dvars = domain_obj.effective_design_vars()
        cons  = wo.constraint_funcs_from_terms(domain_obj.evaluator, dvars)
        return DifferentialEvolutionOptimizer(domain_obj.evaluator, wo, cons, dvars,
                                              de_seed=de_seed)
    return domain_obj.optimizer


def get_feasibility_repair(domain_obj: HardwareDomain) -> FeasibilityRepair:
    return FeasibilityRepair(
        constraint_funcs=domain_obj.constraint_funcs,
        design_vars=domain_obj.effective_design_vars(),
        evaluator=domain_obj.evaluator,
    )

def _sep(char: str = "=", width: int = 68) -> None: print(char * width)


def _print_result(result: OptimResult, generator: AbstractGeometryGenerator,
                  domain_name: str, label: str = "") -> None:
    status = "SUCCEEDED" if result.success else "DID NOT CONVERGE"
    feas   = getattr(result, "feasible", True)
    feas_s = "FEASIBLE" if feas else "⚠ INFEASIBLE"
    tag    = f"  [{label}]" if label else ""
    _sep()
    print(f"  {domain_name.upper()}{tag}  |  {status}  |  {feas_s}  "
          f"({result.n_iter} iters, {result.elapsed_s:.2f}s)")
    _sep()
    print(f"  Objective value : {result.objective_value:.6g}")
    print("\nOptimised design variables:")
    for k, v in result.optimized_params.items():
        if isinstance(v, (int, float)):
            print(f"  {k:<36} = {v:.6g}")
    print("\nPerformance metrics:")
    for k, v in result.scalar_results().items():
        print(f"  {k:<36} = {v:.6g}")
    geo = result.results.get("geometry")
    if geo: print(); print(generator.export_geometry(geo))


def run_single(domain_obj: HardwareDomain, domain_name: str,
               optimizer: AbstractOptimizer,
               repair: Optional[FeasibilityRepair] = None) -> OptimResult:
    params = domain_obj.effective_params()
    _sep(); print(f"  Hardware Optimiser v3 – {domain_name.upper()}  [single]"); _sep()
    print("Effective initial parameters:")
    for k, v in params.items(): print(f"  {k:<36} = {v}")
    print()
    result = optimizer.optimize_design(params)
    if repair is not None:
        result = repair.repair(result)
    _print_result(result, domain_obj.geometry_generator, domain_name, label="single")
    _sep(); return result


def run_sampling(domain_obj: HardwareDomain, domain_name: str, optimizer: AbstractOptimizer,
                 mode: SamplingMode, n_runs: int, seed: int,
                 repair: Optional[FeasibilityRepair] = None,
                 max_workers: int = 1) -> OptimResult:
    params = domain_obj.effective_params()
    dvars  = domain_obj.effective_design_vars()
    _sep(); print(f"  Hardware Optimiser v3 – {domain_name.upper()}  "
                  f"[{mode}  n={n_runs}  seed={seed}  workers={max_workers}]"); _sep()
    ms = run_multistart(optimizer, params, dvars, mode=mode, n_runs=n_runs, seed=seed,
                        feasibility_repair=repair, max_workers=max_workers)
    print(f"\n{ms.summary_table()}\n")
    print("── GLOBAL BEST " + "─" * 53)
    _print_result(ms.best, domain_obj.geometry_generator, domain_name, label="best")
    _sep(); return ms.best


def run_differential_evo(domain_obj: HardwareDomain, domain_name: str,
                         optimizer: DifferentialEvolutionOptimizer,
                         repair: Optional[FeasibilityRepair] = None) -> OptimResult:
    params = domain_obj.effective_params()
    _sep(); print(f"  Hardware Optimiser v3 – {domain_name.upper()}  [differential_evolution]"); _sep()
    result = optimizer.optimize_design(params)
    if repair is not None:
        result = repair.repair(result)
    _print_result(result, domain_obj.geometry_generator, domain_name, label="DE+polish")
    _sep(); return result


def run_tolerance_study(best_result: OptimResult, domain_obj: HardwareDomain,
                        domain_name: str, n_samples: int, seed: int,
                        config_tols: Dict[str, float]) -> None:
    _sep("-"); print(f"  Tolerance MC – {domain_name.upper()}  (n={n_samples}, seed={seed})"); _sep("-")
    tols_all   = {**domain_obj.default_tols(), **config_tols}
    tol_params = {k: tols_all[k] for k in domain_obj.evaluator.tolerance_params if k in tols_all}
    if not tol_params:
        log.warning("No tolerance params resolved for '%s'. Skipping.", domain_name); return
    mc = run_tolerance_mc(
        best_params=best_result.optimized_params, evaluator=domain_obj.evaluator,
        constraint_funcs=domain_obj.constraint_funcs, design_vars=domain_obj.effective_design_vars(),
        tols=tol_params, n_samples=n_samples, seed=seed)
    print(mc.report()); _sep("-")

def _run_tests() -> None:
    """
    11 high-value unit tests.  No pytest required; run with --test.
    Covers physics functions, gradients, feasibility repair, mass models,
    weighted objective framework, and DomainConfig plumbing.
    """
    passed: List[str] = []
    failed: List[str] = []

    def _test(name: str, fn: Callable[[], None]) -> None:
        try:
            fn()
            passed.append(name); print(f"  PASS  {name}")
        except Exception as exc:
            failed.append(name); print(f"  FAIL  {name}: {exc}")
            _tb.print_exc()

    print("\n" + "="*60 + "\n  Unit Tests  (v3)\n" + "="*60)

    # T1 – Nozzle exit velocity: known isentropic formula
    def _t1() -> None:
        subs = {"gamma": 1.4, "R": 287.0, "T0": 300.0, "Me": 2.0}
        Ve       = _nozzle_exit_velocity(subs)
        expected = 2.0 * math.sqrt(1.4 * 287.0 * 300.0 / (1.0 + 0.2 * 4.0))
        assert abs(Ve - expected) < 0.05, f"Ve={Ve:.3f}  expected={expected:.3f}"

    # T2 – Nozzle area-ratio Brent solver round-trip
    def _t2() -> None:
        gamma, Me_in = 1.4, 2.5
        gp1, gm1 = gamma+1.0, gamma-1.0
        ar = (1.0/Me_in)*((2.0/gp1)*(1.0+0.5*gm1*Me_in**2))**(0.5*gp1/gm1)
        Me_out = _nozzle_exit_mach({"gamma": gamma, "area_ratio": ar})
        assert abs(Me_out - Me_in) < 1e-6, f"Round-trip err={abs(Me_out-Me_in):.2e}"

    # T3 – PMSM torque: direct manual check
    def _t3() -> None:
        subs = {"pole_pairs": 4.0, "flux_linkage_pm": 0.05, "current_q": 100.0,
                "rotor_radius": 0.07}
        T = _pmsm_torque(subs)
        assert abs(T - 30.0) < 1e-9, f"Torque={T}  expected=30.0"

    # T4 – PMSM power density: energy chain consistency
    def _t4() -> None:
        subs = {"pole_pairs":4.0,"rotor_radius":0.07,"stack_length":0.12,
                "B_gap":0.85,"winding_factor":0.933,"turns_per_phase":12.0,
                "current_q":200.0,"omega_mech":300.0,"mass_total":10.0}
        psi  = _pmsm_flux_linkage(subs)
        subs2 = {**subs, "flux_linkage_pm": psi}
        T    = _pmsm_torque(subs2)
        subs3 = {**subs2, "torque": T}
        P    = _pmsm_power(subs3)
        subs4 = {**subs3, "power": P}
        pd   = _pmsm_power_density(subs4)
        assert abs(pd - P/1000.0/10.0) < 1e-9, f"PD chain error: pd={pd}"

    # T5 – Pump tip speed formula
    def _t5() -> None:
        subs = {"rotational_speed_rpm": 1450.0, "impeller_diameter_m": 0.25}
        expected = math.pi * 1450.0 * 0.25 / 60.0
        assert abs(_pump_tip_speed(subs) - expected) < 1e-9

    # T6 – Pump head plausibility for default params
    def _t6() -> None:
        p = PumpParams()
        s = p.to_dict()
        s["tip_speed"]       = _pump_tip_speed(s)
        s["slip_factor"]     = _pump_slip_factor(s)
        s["radial_velocity"] = _pump_radial_velocity(s)
        H = _pump_head(s)
        assert 5.0 < H < 800.0, f"Head implausible: {H:.1f} m"

    # T7 – WeightedObjective: at nominal → zero; above nominal improves objective
    def _t7() -> None:
        terms = [ObjectiveTerm("eff", -1.0, 0.8)]
        wo = WeightedObjective(terms, {})
        assert wo({"eff": 0.8}) == 0.0, "Nominal must give zero"
        assert wo({"eff": 1.0}) < wo({"eff": 0.5}), "Higher eff must give lower obj"

    # T8 – DomainConfig bound override flows through effective_design_vars
    def _t8() -> None:
        cfg = DomainConfig(bounds={"impeller_diameter_m": (0.10, 0.60)})
        dvars = dict(CentrifugalPumpDomain(cfg).effective_design_vars())
        lo, hi = dvars["impeller_diameter_m"]
        assert abs(lo-0.10)<1e-9 and abs(hi-0.60)<1e-9, f"Got ({lo},{hi})"

    # T9 – Analytical gradient of specific_impulse wrt Me (finite-diff check)
    def _t9() -> None:
        params = NozzleParams().to_dict()
        dvars  = ["Me", "throat_radius"]
        g_anal = _nozzle_grad_specific_impulse(params, dvars)
        eps    = 1e-6
        p_plus  = {**params, "Me": params["Me"] + eps}
        p_minus = {**params, "Me": params["Me"] - eps}
        # Evaluate Isp numerically
        def _isp(p):
            gamma, R, T0, Me = p["gamma"], p["R"], p["T0"], p["Me"]
            Te = T0 / (1.0 + 0.5*(gamma-1.0)*Me**2)
            Ve = Me * math.sqrt(gamma * R * Te)
            return Ve / 9.80665
        fd = (_isp(p_plus) - _isp(p_minus)) / (2 * eps)
        rel_err = abs(g_anal[0] - fd) / (abs(fd) + 1e-30)
        assert rel_err < 1e-4, f"Isp grad rel_err={rel_err:.2e}"

    # T10 – FeasibilityRepair: infeasible point projected to feasible region
    def _t10() -> None:
        # Simple 1D problem: minimise x, constraint x >= 0.5, bound [0,1]
        cfg       = DomainConfig.empty()
        domain    = NozzleDomain(cfg)
        dvars     = [("Me", (2.0, 5.0)), ("throat_radius", (0.02, 0.10))]
        # Create a dummy infeasible result (Me=1.8 violates Me>=2.0 lower bound)
        p0 = {**domain.default_params, "Me": 1.8}
        ev = domain.evaluator
        res0 = OptimResult(
            optimized_params=p0, results=ev.evaluate({**p0, "Me": 2.0}),
            objective_value=99.0, success=False, message="test", n_iter=0, elapsed_s=0.0,
            feasible=False)
        repair = get_feasibility_repair(domain)
        # Clamp to bounds manually first to ensure evaluator works
        p0["Me"] = 2.0
        res0 = OptimResult(
            optimized_params=p0, results=ev.evaluate(p0),
            objective_value=99.0, success=True, message="test", n_iter=0, elapsed_s=0.0)
        repaired = repair.repair(res0)
        assert repaired.feasible, "Repaired result should be feasible"

    # T11 – MassModelRegistry: table interpolation
    def _t11() -> None:
        cfg_dict = {
            "nozzle": {
                "model_type": "table",
                "x_key": "throat_radius",
                "x": [0.02, 0.05, 0.10],
                "mass_kg": [1.0, 3.0, 8.0],
            }
        }
        reg  = MassModelRegistry("nozzle", cfg_dict)
        mass = reg.calculate({"throat_radius": 0.05}, lambda p: 999.0)
        assert abs(mass - 3.0) < 1e-9, f"Table interp: {mass}"
        mass2 = reg.calculate({"throat_radius": 0.075}, lambda p: 999.0)
        assert abs(mass2 - 5.5) < 1e-9, f"Table interp mid: {mass2}"

    for name, fn in [
        ("T1  Nozzle exit velocity",              _t1),
        ("T2  Nozzle area-ratio round-trip",      _t2),
        ("T3  PMSM torque formula",               _t3),
        ("T4  PMSM power density chain",          _t4),
        ("T5  Pump tip speed formula",            _t5),
        ("T6  Pump head plausibility",            _t6),
        ("T7  WeightedObjective normalise",       _t7),
        ("T8  DomainConfig bound override",       _t8),
        ("T9  Nozzle Isp analytical gradient",    _t9),
        ("T10 FeasibilityRepair round-trip",      _t10),
        ("T11 MassModelRegistry table interp",    _t11),
    ]:
        _test(name, fn)

    print("="*60)
    print(f"  {len(passed)} passed  |  {len(failed)} failed")
    print("="*60 + "\n")
    if failed:
        raise SystemExit(1)

                        help="Disable analytical gradients (use FD only).")
    parser.add_argument("--dashboard", action="store_true",
                        help="Launch the Streamlit dashboard.")
    parser.add_argument("--test",     action="store_true", help="Run unit tests and exit.")
    parser.add_argument("--verbose",  action="store_true")
    args = parser.parse_args(argv)

    if args.verbose:
        logging.getLogger().setLevel(logging.DEBUG)

    if args.test:
        _run_tests(); return

    if args.dashboard:
        _launch_dashboard(); return

    cfg        = DomainConfig.from_file_or_empty(args.config)
    domain_obj = get_domain(args.domain, cfg)
    optimizer  = get_optimizer(domain_obj, args.mode, de_seed=args.seed)
    repair     = None if args.no_repair else get_feasibility_repair(domain_obj)
    workers    = args.workers if args.parallel else 1

    if args.mode == "single":
        best = run_single(domain_obj, args.domain, optimizer, repair=repair)
    elif args.mode == "differential_evo":
        best = run_differential_evo(domain_obj, args.domain, optimizer, repair=repair)
    else:
        best = run_sampling(domain_obj, args.domain, optimizer,
                            mode=args.mode, n_runs=args.n_runs, seed=args.seed,
                            repair=repair, max_workers=workers)

    if args.tolerance_mc > 0:
        run_tolerance_study(best, domain_obj, args.domain,
                            n_samples=args.tolerance_mc, seed=args.seed, config_tols=cfg.tols)

def _launch_dashboard() -> None:
    """Launch the Streamlit UI in a subprocess."""
    import subprocess, sys, os

    script_dir = Path(__file__).resolve().parent
    dash_path  = script_dir / "hardware_optimiser_dashboard.py"
    _write_dashboard_script(dash_path)
    print(f"Launching Streamlit dashboard: {dash_path}")
    subprocess.run(
        [sys.executable, "-m", "streamlit", "run", str(dash_path),
         "--server.headless", "false"],
        check=False)
